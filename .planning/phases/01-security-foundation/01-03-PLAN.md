## Step 5 — Add .angel directory initializer

Create `src/utils/angelDir.ts`:
```typescript
import * as fs from 'fs';
import * as path from 'path';

export function ensureAngelDir(workspaceRoot: string): string {
    const angelDir = path.join(workspaceRoot, '.angel');
    fs.mkdirSync(angelDir, { recursive: true });
    return angelDir;
}

export function angelPath(workspaceRoot: string, ...segments: string[]): string {
    return path.join(workspaceRoot, '.angel', ...segments);
}
```

Use `angelPath()` in all agents instead of string template literals.
